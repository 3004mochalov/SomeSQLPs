    CREATE TABLE messages (
        id SERIAL PRIMARY KEY,
        content VARCHAR(255),
        created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
    );
CREATE TABLE users (
    id SERIAL PRIMARY KEY,
    username VARCHAR(100) UNIQUE NOT NULL,
    password TEXT NOT NULL
);
CREATE TABLE scores (
    id SERIAL PRIMARY KEY,
    username VARCHAR(100) NOT NULL,
    score INTEGER NOT NULL
);
